|---------------------------------------------------------------
| LOCATION HELPER
|---------------------------------------------------------------
|
| This is a new helper created for SMS 3 that aids in find the
| location of an image or view file. There are 2 functions associated 
| with this helper.
|
| view_location($view_file, $theme, $section)
| img_location($image, $theme, $section)
| js_location($js_file, $theme, $section)
|

VIEW LOCATION
==========

@ param 	string (default = empty string)
@ param 	string (default = default)
@ param 	string (default = empty string)

@ autoload	yes
@ notes		checks the views folder for a directory named after param2,
			if it doesn't find it, rolls over to GLOBAL

@ return 	string (param output)

EXAMPLE:

$view_loc = view_location('main_index', $this->session->userdata('theme_main'), 'main');
echo $view_loc;

OUTPUT:

default/main/pages/main_index

IMAGE LOCATION
==========

@ param 	string (default = empty string)
@ param 	string (default = empty string)
@ param 	string (default = empty string)

@ autoload	yes
@ notes		checks the views folder for a directory named after param2,
			if it doesn't find it, rolls over to GLOBAL

@ return 	string (param output)

EXAMPLE:

$image = img_location('saved.png', $this->session->userdata('theme_main'), 'main');
echo $image;

OUTPUT:

default/main/img/saved.png

JS LOCATION
==========

@ param 	string (default = empty string)
@ param 	string (default = default)
@ param 	string (default = empty string)

@ autoload	yes
@ notes		checks the views folder for a directory named after param2,
			if it doesn't find it, rolls over to GLOBAL; if the file
			doesn't exist in GLOBAL, returns FALSE

@ return 	string (param output)

EXAMPLE:

$data['js'] = js_location('main_index', $this->session->userdata('theme_main'), 'main');
echo $data['js'];

OUTPUT:

global/main/js/main_index.js

ASSET LOCATION
==========

@ param 	string (default = empty string)
@ param 	string (default = empty string)
@ param		string (default = {application folder})

@ autoload	yes
@ notes		checks the assets/common folder for the directory of
			param1 and the file of param2

@ return 	string (param output)

EXAMPLE:

$asset = asset_location(GENRE .'/img/', 'image.png', APPFOLDER);
echo $asset;

OUTPUT:

application/assets/common/bsg/img/image.png
