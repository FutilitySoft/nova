|---------------------------------------------------------------
| SYSTEM MODEL
|
| System Version: 1.0
|---------------------------------------------------------------

check_install_status()

Parameters		=>	none
Returns			=>	TRUE/FALSE
Description		=>	checks to see whether the system is installed or not

==========
	
get_all_ranks()

Parameters		=>	status (*active*, inactive, development, NULL)
Returns			=>	query object
Description		=>	grabs all the rank sets from the rank catalogue

==========
	
get_all_skins()

Parameters		=>	none
Returns			=>	query object
Description		=>	grabs all the skins from the skin catalogue

==========
	
get_current_version()

Parameters		=>	none
Returns			=>	complete system version (1.0.0)
Description		=>	pulls the complete system version from the system table

==========
	
get_database_size()

Parameters		=>	none
Returns			=>	size of the database
Description		=>	calculates the size of the database

==========
	
get_nova_uid()

Parameters		=>	none
Returns			=>	the system ID if it exists or FALSE otherwise
Description		=>	grabs Nova's unique identifier

==========
	
get_rank_info()

Parameters		=>	location of the rank set
Returns			=>	query object row if there is one or FALSE otherwise
Description		=>	pulls the row from the rank catalogue for the location passed to the function

==========
	
get_security_questions()

Parameters		=>	none
Returns			=>	query object
Description		=>	grabs all the security questions from the database

==========
	
get_sim_types()

Parameters		=>	none
Returns			=>	query object
Description		=>	grabs all the sim types from the database

==========
	
get_skin_info()

Parameters		=>	location of the skin
Returns			=>	query object row if there is one or FALSE otherwise
Description		=>	pulls the row from the skin catalogue for the location passed to the function

==========
	
get_skin_name()

Parameters		=>	ID of a skin
Returns			=>	skin name if the row exists or FALSE otherwise
Description		=>	pulls the skin name from the skin catalogue

==========
	
get_skin_themes()

Parameters		=>	ID of a skin
					status of the skin (*active*, inactive, development, NULL)
Returns			=>	query object
Description		=>	grabs all the themes associated with a given skin ID

==========

get_whatsnew()

Parameters		=>	none
Returns			=>	what's new field from system table
Description		=>	grabs the what's new information from the database

==========	
	
add_rank_set()

Parameters		=>	array or object with data to be inserted into the database
Returns			=>	number of affected rows (should be 1)
Description		=>	inserts a rank set into the rank catalogue table

==========
	
update_database_charset()

Parameters		=>	none
Returns			=>	number of affected rows
Description		=>	updates the database charset

==========
	
update_my_links()

Parameters		=>	none
Returns			=>	number of affected rows
Description		=>	updates all players to have a preset my_links field
